def allowed_file(filename):
    """Check if file extension is allowed."""
    return '.' in filename and filename.rsplit('.', 1)[1].lower() in ALLOWED_EXTENSIONS

@app.route('/analyze', methods=['POST'])
def analyze_file():
    """
    Analyze uploaded CSV file with Explainable AI.
    IMPROVED: Better error messages and validation
    """
    if detector is None:
        return jsonify({
            'error': 'Detection system not initialized',
            'details': 'The AI model could not be loaded. Please check server logs.',
            'suggestion': 'Ensure model files are present in the correct directory.'
        }), 500
    
    # Validate file upload
    if 'file' not in request.files:
        return jsonify({
            'error': 'No file uploaded',
            'details': 'The request did not contain a file.',
            'suggestion': 'Please select a CSV file to analyze.'
        }), 400
    
    file = request.files['file']
    
    if file.filename == '':
        return jsonify({
            'error': 'No file selected',
            'details': 'File field was empty.',
            'suggestion': 'Please select a CSV file before clicking Analyze.'
        }), 400
    
    if not allowed_file(file.filename):
        return jsonify({
            'error': 'Invalid file type',
            'details': f'File "{file.filename}" is not a CSV file.',
            'suggestion': 'Only .csv files are supported. Please upload a CSV file.',
            'file_received': file.filename
        }), 400
    
    # Get parameters
    model_name = request.form.get('model', 'xgboost')
    try:
        explain_samples = int(request.form.get('explain_samples', 5))
        if explain_samples < 1 or explain_samples > 20:
            explain_samples = 5
    except ValueError:
        explain_samples = 5
    
    # Validate model
    if model_name not in detector.models:
        return jsonify({
            'error': f'Invalid model: {model_name}',
            'available_models': list(detector.models.keys()),
            'suggestion': f'Please select one of: {", ".join(detector.models.keys())}'
        }), 400
    
    temp_path = None
    try:
        # Save uploaded file
        filename = secure_filename(file.filename)
        timestamp = datetime.now().strftime('%Y%m%d_%H%M%S')
        temp_path = os.path.join(
            app.config['UPLOAD_FOLDER'], 
            f"nids_{timestamp}_{filename}"
        )
        file.save(temp_path)
        
        # Check file size
        file_size_mb = os.path.getsize(temp_path) / (1024 * 1024)
        
        print(f"\n{'='*80}")
        print(f"Processing: {filename}")
        print(f"File size: {file_size_mb:.2f} MB")
        print(f"Model: {model_name}")
        print(f"XAI samples: {explain_samples}")
        print(f"{'='*80}\n")
        
        # Validate CSV can be read
        try:
            import pandas as pd
            test_df = pd.read_csv(temp_path, nrows=5)
            print(f"✓ CSV validated: {len(test_df.columns)} columns detected")
            del test_df
        except Exception as csv_error:
            return jsonify({
                'error': 'Invalid CSV file',
                'details': str(csv_error),
                'suggestion': 'The file appears to be corrupted or not a valid CSV. Please check the file format.'
            }), 400
        
        # Analyze with XAI
        output_dir = os.path.join(app.config['UPLOAD_FOLDER'], 'xai_results')
        os.makedirs(output_dir, exist_ok=True)
        
        results = detector.analyze_file(
            csv_file_path=temp_path,
            model_name=model_name,
            explain_samples=explain_samples,
            save_results=True,
            output_dir=output_dir
        )
        
        # Clean up temp file
        try:
            if temp_path and os.path.exists(temp_path):
                os.remove(temp_path)
        except Exception as cleanup_error:
            print(f"⚠️  Could not delete temp file: {cleanup_error}")
        
        # Check for errors in results
        if 'error' in results:
            return jsonify({
                'error': 'Analysis failed',
                'details': results['error'],
                'suggestion': 'The file may have an incompatible format or missing required columns.',
                'file_info': {
                    'name': filename,
                    'size_mb': round(file_size_mb, 2)
                }
            }), 500
        
        print(f"\n✅ Analysis complete!")
        print(f"   Processed: {results['total_samples']:,} samples")
        print(f"   Attacks: {results['attack_count']:,} ({results['attack_percentage']}%)")
        print(f"   Benign: {results['benign_count']:,} ({results['benign_percentage']}%)")
        print(f"{'='*80}\n")
        
        # Add file info to results
        results['file_info'] = {
            'name': filename,
            'size_mb': round(file_size_mb, 2),
            'analysis_timestamp': datetime.now().isoformat()
        }
        
        return jsonify(results)
    
    except MemoryError:
        # Clean up on memory error
        if temp_path and os.path.exists(temp_path):
            try:
                os.remove(temp_path)
            except:
                pass
        
        return jsonify({
            'error': 'Out of memory',
            'details': 'The file is too large to process with available memory.',
            'suggestion': 'Try a smaller file or reduce the number of samples to explain.',
            'file_size_mb': round(file_size_mb, 2) if 'file_size_mb' in locals() else 'unknown'
        }), 507
    
    except Exception as e:
        # Clean up on error
        if temp_path and os.path.exists(temp_path):
            try:
                os.remove(temp_path)
            except:
                pass
        
        print(f"\n❌ Analysis Error: {e}")
        traceback.print_exc()
        
        error_message = str(e)
        
        # Provide helpful suggestions based on error type
        suggestion = 'Please check the file format and try again.'
        
        if 'UnicodeDecodeError' in error_message or 'encoding' in error_message.lower():
            suggestion = 'The file has encoding issues. Try saving it as UTF-8 encoded CSV.'
        elif 'KeyError' in error_message or 'column' in error_message.lower():
            suggestion = 'Some expected columns may be missing. Ensure your CSV has network traffic features.'
        elif 'memory' in error_message.lower():
            suggestion = 'File is too large. Try a smaller dataset or reduce explain_samples parameter.'
        
        return jsonify({
            'error': 'Analysis failed',
            'details': error_message,
            'suggestion': suggestion,
            'error_type': type(e).__name__
        }), 500
