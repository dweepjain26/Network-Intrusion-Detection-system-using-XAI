// Create floating particles
function createParticles() {
    const particlesContainer = document.getElementById('particles');
    for (let i = 0; i < 50; i++) {
        const particle = document.createElement('div');
        particle.className = 'particle';
        particle.style.left = Math.random() * 100 + '%';
        particle.style.animationDelay = Math.random() * 20 + 's';
        particle.style.animationDuration = (15 + Math.random() * 10) + 's';
        particlesContainer.appendChild(particle);
    }
}

// Initialize
async function init() {
    try {
        const response = await fetch('http://localhost:5000/models');
        const data = await response.json();
        
        availableModels = data.models;
        populateModelSelect(data.models);
        
        showAlert(`System ready! ${data.models.length} models available.`, 'success', '✅');
    } catch (error) {
        showAlert('Warning: Could not load models. Using defaults.', 'warning', '⚠️');
        populateModelSelect([
            {key: 'xgboost', name: 'XGBoost', recommended: true},
            {key: 'random_forest', name: 'Random Forest', recommended: false}
        ]);
    }
}

function populateModelSelect(models) {
    modelSelect.innerHTML = '';
    
    models.forEach(model => {
        const option = document.createElement('option');
        option.value = model.key;
        
        let label = model.name || model.key;
        if (model.recommended) {
            label += ' ⭐';
        }
        
        option.textContent = label;
        option.title = model.description || '';
        
        modelSelect.appendChild(option);
    });
}

dropZone.addEventListener('click', () => fileInput.click());

    if (e.target.files.length > 0) {
        handleFileSelect(e.target.files[0]);
    }
});

    dropZone.classList.remove('drag-over');

    }

function handleFileSelect(file) {
    if (!file.name.endsWith('.csv')) {
        showAlert('Please select a CSV file', 'error', '❌');
        return;
    }
    selectedFile = file;
    analyzeBtn.disabled = false;
    showAlert(`File loaded: ${file.name} (${(file.size / 1024 / 1024).toFixed(2)} MB)`, 'success', '✅');
}

analyzeBtn.addEventListener('click', analyzeFile);

async function analyzeFile() {
    if (!selectedFile) return;

    loading.classList.add('active');
    results.classList.remove('active');
    alert.classList.remove('active');

    const formData = new FormData();
    formData.append('file', selectedFile);
    formData.append('model', modelSelect.value);
    formData.append('explain_samples', explainSamples.value);

    try {
        document.getElementById('loadingProgress').textContent = 'Uploading and preprocessing...';
        
        const response = await fetch('http://localhost:5000/analyze', {
            method: 'POST',
            body: formData
        });

        document.getElementById('loadingProgress').textContent = 'Processing results...';

        if (!response.ok) {
            const errorData = await response.json();
            throw new Error(errorData.error || 'Analysis failed');
        }

        analysisResults = await response.json();
        
        if (analysisResults.error) {
            throw new Error(analysisResults.error);
        }

        displayResults(analysisResults);
    } catch (error) {
        showAlert('Error: ' + error.message, 'error', '❌');
        console.error('Analysis error:', error);
    } finally {
        loading.classList.remove('active');
    }
}

function showAlert(message, type, icon) {
    const alertEl = document.getElementById('alert');
    const alertIcon = document.getElementById('alertIcon');
    const alertMessage = document.getElementById('alertMessage');
    
    alertIcon.textContent = icon || '⚠️';
    alertMessage.textContent = message;
    alertEl.className = `alert ${type} active`;
    
    const duration = type === 'error' ? 10000 : 5000;
    setTimeout(() => alertEl.classList.remove('active'), duration);
}

function displayResults(data) {
    results.classList.add('active');

    // Basic stats
    document.getElementById('totalSamples').textContent = data.total_samples.toLocaleString();
    document.getElementById('processingTime').textContent = data.processing_time_seconds + 's';
    document.getElementById('processingSpeed').textContent = Math.round(data.samples_per_second).toLocaleString() + '/sec';
    document.getElementById('modelUsed').textContent = data.model_used.toUpperCase();

    // Traffic distribution
    document.getElementById('benignCount').textContent = 
        `${data.benign_count.toLocaleString()} (${data.benign_percentage}%)`;
    document.getElementById('maliciousCount').textContent = 
        `${data.attack_count.toLocaleString()} (${data.attack_percentage}%)`;
    
    setTimeout(() => {
        document.getElementById('benignProgress').style.width = data.benign_percentage + '%';
        document.getElementById('maliciousProgress').style.width = data.attack_percentage + '%';
    }, 100);

    // Confidence
    if (data.average_confidence) {
        const confidencePct = (data.average_confidence * 100).toFixed(2);
        document.getElementById('avgConfidence').textContent = confidencePct + '%';
        setTimeout(() => {
            document.getElementById('confidenceProgress').style.width = confidencePct + '%';
        }, 100);
    }

    // Accuracy
    if (data.validation && data.validation.accuracy) {
        const accuracyRow = document.getElementById('accuracyRow');
        accuracyRow.style.display = 'flex';
        document.getElementById('accuracy').textContent = (data.validation.accuracy * 100).toFixed(2) + '%';
    }

    // Attack types
    displayAttackTypes(data.detection_summary);

    // XAI Explanations
    if (data.explanations) {
        displayLIMEExplanations(data.explanations.lime_explanations);
        displaySHAPExplanation(data.explanations.shap_explanation);
    }

    showAlert('Analysis complete! 🎉', 'success', '✅');
}

function displayAttackTypes(summary) {
    const attackList = document.getElementById('attackList');
    attackList.innerHTML = '';

    const sorted = Object.entries(summary).sort((a, b) => b[1].count - a[1].count);

    sorted.forEach(([attackName, data]) => {
        const isBenign = attackName.toUpperCase().includes('BENIGN') || 
                        attackName.toUpperCase().includes('NORMAL');
        
        const item = document.createElement('div');
        item.className = `attack-item ${isBenign ? 'benign' : 'malicious'}`;
        
        item.innerHTML = `
            <div class="attack-content">
                <div class="attack-name">
                    <span>${isBenign ? '✅' : '⚠️'}</span>
                    <span>${attackName}</span>
                </div>
                <div class="progress-bar" style="max-width: 400px;">
                    <div class="progress-fill" style="width: ${data.percentage}%; background: ${isBenign ? 'linear-gradient(90deg, var(--success), #6ee7b7)' : 'linear-gradient(90deg, var(--danger), #fca5a5)'};"></div>
                </div>
            </div>
            <div class="attack-stats">
                <div class="attack-percentage">${data.percentage}%</div>
                <div class="attack-count">${data.count.toLocaleString()} samples</div>
            </div>
        `;
        
        attackList.appendChild(item);
    });
}

function displayLIMEExplanations(explanations) {
    const container = document.getElementById('limeExplanations');
    container.innerHTML = '';

    if (!explanations || explanations.length === 0) {
        container.innerHTML = '<p style="color: var(--text-tertiary);">No LIME explanations available.</p>';
        return;
    }

    explanations.forEach((exp) => {
        const div = document.createElement('div');
        div.className = 'lime-explanation';
        
        let topFeaturesHtml = '';
        let features = [];
        
        if (exp.top_features && typeof exp.top_features === 'object') {
            if (Array.isArray(exp.top_features)) {
                features = exp.top_features;
            } else {
                const classNames = Object.keys(exp.top_features);
                if (classNames.length > 0) {
                    features = exp.top_features[classNames[0]];
                }
            }
        }
        
        if (!Array.isArray(features)) {
            features = [];
        }
        
        if (features.length === 0) {
            topFeaturesHtml = '<p style="color: var(--text-tertiary);">No feature importance data available</p>';
        } else {
            const maxImp = Math.max(...features.map(f => Math.abs(f.contribution || f.importance || 0)));
            
            features.slice(0, 8).forEach(f => {
                const contrib = f.contribution || f.importance || 0;
                const absImp = Math.abs(contrib);
                const width = maxImp > 0 ? (absImp / maxImp * 100).toFixed(1) : 0;
                const gradient = contrib > 0 
                    ? 'linear-gradient(90deg, var(--success), #6ee7b7)' 
                    : 'linear-gradient(90deg, var(--danger), #fca5a5)';
                const featureName = f.feature || f.name || 'Unknown';
                
                topFeaturesHtml += `
                    <div class="feature-bar">
                        <span class="feature-name" title="${featureName}">${featureName}</span>
                        <div class="feature-value" style="width: ${width}%; background: ${gradient};"></div>
                        <span class="feature-score">${contrib.toFixed(4)}</span>
                    </div>
                `;
            });
        }

        div.innerHTML = `
            <div class="lime-header">
                <span>📌</span>
                <h4>Sample #${exp.instance_index} - ${exp.predicted_class || exp.predicted_label || 'Unknown'}</h4>
            </div>
            <p class="lime-confidence">
                Confidence: <span class="confidence-badge">${exp.confidence ? (exp.confidence * 100).toFixed(2) : 'N/A'}%</span>
            </p>
            <div>
                <div class="feature-importance-header">Top Contributing Features</div>
                ${topFeaturesHtml}
            </div>
            ${exp.visualization ? `<img src="data:image/png;base64,${exp.visualization}" class="xai-viz" alt="LIME visualization" />` : ''}
        `;
        
        container.appendChild(div);
    });
}

function displaySHAPExplanation(explanation) {
    const container = document.getElementById('shapExplanation');
    container.innerHTML = '';

    if (!explanation) {
        container.innerHTML = '<p style="color: var(--text-tertiary);">No SHAP explanation available.</p>';
        return;
    }

    let featuresHtml = '';
    const features = Object.entries(explanation.top_features);
    const maxImp = Math.max(...features.map(([_, imp]) => imp));
    
    features.forEach(([feature, importance]) => {
        const width = (importance / maxImp * 100).toFixed(1);
        featuresHtml += `
            <div class="feature-bar">
                <span class="feature-name" title="${feature}">${feature}</span>
                <div class="feature-value" style="width: ${width}%; background: linear-gradient(90deg, var(--primary), var(--primary-light));"></div>
                <span class="feature-score">${importance.toFixed(4)}</span>
            </div>
        `;
    });

    container.innerHTML = `
        <div style="margin-bottom: 24px;">
            <div class="feature-importance-header">Global Feature Importance (Top 10 Features)</div>
            <p style="color: var(--text-secondary); margin: 10px 0; font-size: 13px;">
                Based on ${explanation.num_samples_explained} samples analyzed
            </p>
            <div style="margin-top: 16px;">
                ${featuresHtml}
            </div>
        </div>
        ${explanation.visualization ? `<img src="data:image/png;base64,${explanation.visualization}" class="xai-viz" alt="SHAP visualization" />` : ''}
    `;
}

function switchTab(tab, event) {
    document.querySelectorAll('.tab').forEach(t => t.classList.remove('active'));
    document.querySelectorAll('.tab-content').forEach(c => c.classList.remove('active'));
    
    event.currentTarget.classList.add('active');
    document.getElementById(tab + '-tab').classList.add('active');
}
